// STM32 MAIN BOARD TO ESP32

PA2(USART 2 TX) --> D16(UART 2 RX)
PA3(USART 2 RX) --> D17(UART 2 TX)
GND ---> GND


//// IMPORTANT ::::: 120 OHM TERMINATING RESISTOR BETWEEN CANH AND CANL OF BOTH TRANSRECEIVER ::::: //// 

// CAN TRANSRECEIVER TO STM32 PILL

VCC --> 5V
GND --> GND
CTX --> PB9 (CAN_TX)
CRX --> PB8 (CAN_RX)
CANH --> CANH
CANL --> CANL

// CAN TRANSRECEIVER TO STM32 MAIN BOARD

VCC --> 5V
GND --> GND
CTX --> PB9 (CAN_TX)
CRX --> PB8 (CAN_RX)
CANH --> CANH
CANL --> CANL
